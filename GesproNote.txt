-- Création et utilisation de la base de données
CREATE DATABASE IF NOT EXISTS Gespro CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE Gespro;

-- 1. Table Gestionnaires (Écran de connexion)
CREATE TABLE gestionnaires (
    id_gestionnaire INT AUTO_INCREMENT PRIMARY KEY,
    nom_complet VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    mot_de_passe VARCHAR(255) NOT NULL, -- Stocker le mot de passe haché (rôle de Nathan)
    cree_le TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 2. Table Niveaux (Frais d'inscription variables)
CREATE TABLE niveaux (
    id_niveau INT AUTO_INCREMENT PRIMARY KEY,
    nom_niveau VARCHAR(50) NOT NULL, -- Ex: Licence 1, Licence 2, Master 1
    montant_inscription DECIMAL(10,2) NOT NULL
) ENGINE=InnoDB;

-- 3. Table Etudiants
CREATE TABLE etudiants (
    id_etudiant INT AUTO_INCREMENT PRIMARY KEY,
    matricule VARCHAR(20) NOT NULL UNIQUE,
    nom VARCHAR(50) NOT NULL,
    prenom VARCHAR(50) NOT NULL,
    sexe ENUM('M', 'F') NOT NULL,
    filiere VARCHAR(100) NOT NULL, -- La classe / filière (Texte)
    cycle VARCHAR(50) NOT NULL,    -- Le cycle (Texte : ex: Premier cycle, Master)
    adresse VARCHAR(255),
    telephone VARCHAR(20),
    email VARCHAR(100) UNIQUE,
    id_niveau INT, -- Niveau académique actuel
    statut_inscription ENUM('Inscrit', 'Réinscrit') NOT NULL,
    decision_passage ENUM('Admis', 'Redoublant', 'En cours') DEFAULT 'En cours', -- Gère le passage en classe supérieure
    date_enregistrement TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_niveau) REFERENCES niveaux(id_niveau) ON DELETE SET NULL
) ENGINE=InnoDB;

-- 4. Table Paiements (Modèle basé sur vos notes et le reçu physique)
CREATE TABLE paiements (
    id_paiement INT AUTO_INCREMENT PRIMARY KEY,
    numero_recu VARCHAR(50) NOT NULL UNIQUE, -- Le numéro unique écrit en rouge sur le reçu
    id_etudiant INT NOT NULL,
    id_gestionnaire INT NOT NULL, -- Le gestionnaire ayant enregistré l'argent
    type_frais ENUM('Inscription', 'Mensuel', 'Diplôme', 'Autre') NOT NULL, -- 'Autre' pour motifs libres (ex: tenue scolaire)
    mois_concerne INT NULL, -- De 1 à 10 (Utile uniquement pour les frais mensuels)
    montant DECIMAL(10,2) NOT NULL, -- Montant en chiffres (ex: 5000.00)
    montant_lettres VARCHAR(255) NOT NULL, -- Montant en lettres (ex: 'Cinq mille francs CFA')
    date_paiement DATE NOT NULL, -- Date inscrite manuellement ou automatiquement sur le reçu
    cree_le TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_etudiant) REFERENCES etudiants(id_etudiant),
    FOREIGN KEY (id_gestionnaire) REFERENCES gestionnaires(id_gestionnaire)
) ENGINE=InnoDB;

from reportlab.lib.pagesizes import letter
from reportlab.pdfgen import canvas

def generer_recu_pdf(numero_recu):
    nom_fichier = f"Recu_{numero_recu}.pdf"
    c = canvas.Canvas(nom_fichier, pagesize=letter)
    
    # Dessin du reçu (Inspiré de l'image_52f422.jpg)
    c.setFont("Helvetica-Bold", 16)
    c.drawString(50, 750, "INSTITUT SUPERIEUR POLYTECHNIQUE")
    
    c.setFont("Helvetica", 12)
    c.drawString(400, 750, f"REÇU No: {numero_recu}")
    c.drawString(50, 700, "Reçu de : ONDONGO GAMPIO Ruth")
    c.drawString(50, 680, "Motif : Paiement tenue scolaire")
    c.drawString(50, 660, "Montant : 5 000 FCFA")
    
    # LE LIBELLÉ SIGNATURE (Géré directement dans le code graphique)
    c.setFont("Helvetica-Oblique", 11)
    c.drawString(400, 550, "Signature de la gestionnaire :")
    c.line(400, 500, 550, 500) # Ligne horizontale pour signer
    
    c.save()
